function render(data) { if (data.error) { document.getElementById('match-info').innerText = "Eroare API: " + data.error; return; } var m = data.main; var std = data.standings; // 1. Sidebar (Clasament) var sHtml = ""; for (var i = 0; i < std.length; i++) { sHtml += '' + std[i].position + '' + std[i].team.shortName + '' + std[i].points + ''; if (std[i].team.id === m.homeTeam.id) document.getElementById('h-rank').innerText = "Locul " + std[i].position; if (std[i].team.id === m.awayTeam.id) document.getElementById('a-rank').innerText = "Locul " + std[i].position; } document.getElementById('standings-body').innerHTML = sHtml; // 2. Hero Card document.getElementById('h-logo').src = "https://api.sofascore.app/api/v1/team/" + m.homeTeam.id + "/image"; document.getElementById('a-logo').src = "https://api.sofascore.app/api/v1/team/" + m.awayTeam.id + "/image"; document.getElementById('h-name').innerText = m.homeTeam.shortName; document.getElementById('a-name').innerText = m.awayTeam.shortName; // Verificăm dacă e LIVE sau programat if (m.status.type === "inprogress") { document.getElementById('score').innerText = (m.homeScore.display || 0) + "-" + (m.awayScore.display || 0); document.getElementById('status').className = "status-badge"; // Pulsează roșu } else { var startTime = new Date(m.startTimestamp * 1000).toLocaleTimeString('ro-RO', {hour:'2-digit', minute:'2-digit'}); document.getElementById('score').innerText = startTime; // Arătăm ora în loc de scor document.getElementById('status').className = ""; // Fără animație de puls document.getElementById('status').style.background = "#334155"; } document.getElementById('status').innerText = m.status.description; var startDate = new Date(m.startTimestamp * 1000); document.getElementById('match-info').innerText = startDate.toLocaleDateString('ro-RO', {day:'2-digit', month:'long'}) + " | SuperLiga România"; // 3. Top Cards (Rezultate Etapa Trecută) function formatLast(match, currentId) { if (!match) return "Nu există date"; var h = match.homeTeam; var a = match.awayTeam; var hName = (h.id === currentId) ? ''+h.shortName+'' : h.shortName; var aName = (a.id === currentId) ? ''+a.shortName+'' : a.shortName; return ''+hName+' '+match.homeScore.display+'-'+match.awayScore.display+' '+aName+''; } document.getElementById('h-last-display').innerHTML = formatLast(data.hLast, m.homeTeam.id); document.getElementById('a-last-display').innerHTML = formatLast(data.aLast, m.awayTeam.id); // 4. Bottom Row (Următoarele 3 meciuri) var uHtml = ""; for (var k = 0; k < data.upcoming.length; k++) { var u = data.upcoming[k]; var uD = new Date(u.startTimestamp * 1000); var uTime = uD.toLocaleDateString('ro-RO', {day:'2-digit', month:'short'}) + " " + uD.toLocaleTimeString('ro-RO', {hour:'2-digit', minute:'2-digit'}); uHtml += '
'+uTime+'
'+u.homeTeam.shortName+' vs '+u.awayTeam.shortName+'
'; } document.getElementById('upcoming-list').innerHTML = uHtml; }
